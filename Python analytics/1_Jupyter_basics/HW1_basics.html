def dict_sums(dates: list, sums: list) -> dict:
    d1 = dict.fromkeys(tuple(sorted(set([dates[i][1] for i in range(len(dates))]))), 0)
    for j in range(len(sums)):
        d1[dates[j][1]] += sums[j]
